1- Retrieve the distinct values from the replacement_cost column in the film table.

SELECT DISTINCT replacement_cost FROM film;

2- Count the number of distinct values in the replacement_cost column in the film table.

SELECT COUNT(DISTINCT replacement_cost) AS distinct_count FROM film;

3- Count the number of film titles (title) that start with the character 'T' and have a rating of 'G' in the film table.

SELECT COUNT(*) AS title_count FROM film WHERE title LIKE 'T%' AND rating = 'G';

4- Count the number of country names (country) in the country table that consist of 5 characters.

SELECT COUNT(*) AS country_count FROM country WHERE LENGTH(country) = 5;

5- Count the number of city names in the city table that end with either 'R' or 'r'.

SELECT COUNT(*) AS city_count FROM city WHERE city LIKE '%R' OR city LIKE '%r';
